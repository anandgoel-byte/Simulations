VIVA MCQS RENDERING ENGINE --- function renderQuestions() { const scoreLabel = document.getElementById('viva-score'); Object.keys(vivaData).forEach(qKey => { const data = vivaData[qKey]; const optionsContainer = document.getElementById(`${qKey}-options`); const feedbackContainer = document.getElementById(`${qKey}-feedback`); if (!optionsContainer || !feedbackContainer) return; optionsContainer.innerHTML = ''; // Clear container data.options.forEach((opt, idx) => { // Build Button const btn = document.createElement('button'); btn.className = "text-left p-3 rounded-lg text-[11px] transition-all border leading-snug font-bold bg-white hover:bg-slate-50 border-slate-200 text-slate-700 cursor-pointer w-full shadow-sm flex items-start gap-2"; // Add A, B, C, D labels let letter = idx === 0 ? 'A)' : idx === 1 ? 'B)' : idx === 2 ? 'C)' : 'D)'; btn.innerHTML = `${letter} ${opt.text}`; // Click Logic btn.addEventListener('click', () => { if (appState.answers[qKey] !== null) return; // Prevent double answering appState.answers[qKey] = idx; if (opt.isCorrect) appState.vivaScore++; if (scoreLabel) scoreLabel.textContent = `Score: ${appState.vivaScore} / 4`; // Style all buttons post-click const buttons = optionsContainer.querySelectorAll('button'); buttons.forEach((b, i) => { b.disabled = true; b.classList.remove('bg-white', 'hover:bg-slate-50', 'border-slate-200', 'text-slate-700', 'cursor-pointer', 'shadow-sm'); b.classList.add('cursor-not-allowed', 'shadow-none'); if (appState.answers[qKey] === i) { // The one the user clicked if (data.options[i].isCorrect) b.classList.add('bg-emerald-50', 'border-emerald-400', 'text-emerald-900'); else b.classList.add('bg-rose-50', 'border-rose-400', 'text-rose-900'); } else if (data.options[i].isCorrect) { // Highlight the correct answer faintly if they got it wrong b.classList.add('bg-emerald-50/50', 'border-emerald-300', 'text-emerald-800', 'opacity-70'); } else { // Fade out the rest b.classList.add('bg-slate-50', 'border-slate-200', 'text-slate-400', 'opacity-40'); } }); // Show Feedback feedbackContainer.textContent = opt.exp; feedbackContainer.classList.remove('hidden'); }); optionsContainer.appendChild(btn); }); }); }
